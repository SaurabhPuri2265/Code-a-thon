import java.lang.*;

class Solution {
    public int reverse(int x) {
        long temp = 0;
        while(x != 0) {
            temp = 10 * temp + x % 10;
            x /= 10;
        }
        return (temp > Integer.MAX_VALUE || temp < Integer.MIN_VALUE) ? 0 : (int) temp;
        
    }
}
