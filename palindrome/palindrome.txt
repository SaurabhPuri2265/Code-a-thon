class Solution{
  boolean isPalindrome(int x) {
        String s=Integer.toString(x);
      //System.out.println(s.charAt(0));
      int i=0,j=s.length()-1;
      while(i<=j)
      {
          if(s.charAt(i)!=s.charAt(j))
              return false;
          i++;
          j--;
      }
      
      return true;
        
    }
}
