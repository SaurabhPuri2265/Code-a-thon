/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
        
        if(root==null)
            return null;
        
        return Helper(root,p,q);

    }
    
    public TreeNode Helper(TreeNode root, TreeNode L, TreeNode H)
    {
        if(root==null)return null;
        
        if((L.val<=root.val && root.val<=H.val)|| (H.val<=root.val && root.val<=L.val))
            return root;
        
        if( L.val>root.val && H.val>root.val)
            return Helper(root.right,L,H);
        
        return Helper(root.left,L,H);
    }
}
