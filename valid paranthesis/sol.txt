class Solution {
    public boolean isValid(String s) {
        if(s==null || s.length()==0)
            return true;
        char[] c=s.toCharArray();
        Stack<Character> stack=new Stack<>();
        for(char c1:c)
        {
            if(c1=='('|| c1=='{' || c1=='[')
                stack.push(c1);
            else
            {
                 if (stack.isEmpty())
                    return false;
                if (c1 == ')' && stack.peek() != '(')
                    return false;
                if (c1 == ']' && stack.peek() != '[')
                    return false;
                if (c1 == '}' && stack.peek() != '{')
                    return false;
                
                stack.pop();
            }
        }
        return stack.isEmpty();
        
        
    }
}
